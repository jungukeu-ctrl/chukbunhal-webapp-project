// 메모에서 RPE와 Reps를 더 정밀하게 추출 (지능형 버전 로직 반영) function parseNote(note) { if (!note) return null; const rawNote = String(note); // RPE 추출: rpe9, rpe 8.5 등 let hRpe = NaN; const rpeMatch = rawNote.match(/rpe\s*(\d+(\.\d+)?)/i); if (rpeMatch) hRpe = parseFloat(rpeMatch[1]); // Reps 범위 추출: reps3-6, reps 8~12 등 let minR = null, maxR = null; const repsMatch = rawNote.match(/reps\s*(\d+)\s*[~-]\s*(\d+)/i); if (repsMatch) { minR = parseInt(repsMatch[1]); maxR = parseInt(repsMatch[2]); } else { const singleMatch = rawNote.match(/reps\s*(\d+)/i); if (singleMatch) minR = maxR = parseInt(singleMatch[1]); } return { rpe: hRpe, min: minR, max: maxR }; } // 사용자님이 찾으신 코드의 4단계 그룹핑 로직 + 속도 최적화 통합 function getSmartSuggestion(exName, targetRepsStr, targetRpe) { const targetName = exName.trim().toLowerCase().replace(/\s+/g, ''); const tRpe = parseFloat(String(targetRpe).replace(/[^0-9.]/g, "")); // 목표 범위 설정 let minR = 0, maxR = 999; let rangeMatch = String(targetRepsStr).match(/(\d+)\s*[~-]\s*(\d+)/); if (rangeMatch) { minR = parseInt(rangeMatch[1]); maxR = parseInt(rangeMatch[2]); } else { let val = parseInt(targetRepsStr); if (!isNaN(val)) minR = maxR = val; } const candidates = APP_DATA.history.filter(row => String(row[APP_DATA.colMap.name]).trim().toLowerCase().replace(/\s+/g, '') === targetName ).map(row => { const weight = parseFloat(String(row[APP_DATA.colMap.weight]).replace(/[^0-9.]/g, "")); const reps = parseInt(row[APP_DATA.colMap.reps]); const note = String(row[APP_DATA.colMap.note] || ""); const noteData = parseNote(note); return { weight, reps, note, date: new Date(row[APP_DATA.colMap.date]), hRpe: noteData ? noteData.rpe : NaN, isRpeMatch: (!isNaN(tRpe) && noteData && Math.abs(tRpe - noteData.rpe) < 0.1), isRepMatch: (reps>= minR && reps <= maxR) }; }).filter(c=> !isNaN(c.weight)); if (candidates.length === 0) return { text: "기록 없음", trends: [] }; // 4단계 우선순위 (P2와 P3 순서 바꿈) let group1 = candidates.filter(c => c.isRpeMatch && c.isRepMatch); if (group1.length > 0) return formatResult(group1, "P1 (완전 일치)"); let group2 = candidates.filter(c => c.isRepMatch); // 반복수 우선 if (group2.length > 0) return formatResult(group2, "P2 (Reps 일치)"); let group3 = candidates.filter(c => c.isRpeMatch); // RPE 차선 if (group3.length > 0) return formatResult(group3, "P3 (RPE 일치)"); return formatResult(candidates, "P4 (최근 기록)"); }
